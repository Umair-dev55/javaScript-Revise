Understanding Functions in JavaScript

Functions are reusable blocks of code that perform a specific task. They help in organizing code, improving readability, and reducing redundancy. Functions can take inputs (parameters) and can return outputs (results).

Code Example

// Simple Function
function myFunction() {
    console.log("Umair Ali\nWe are learning js ");
    console.log("we are learning js");
}
myFunction();

// Function with Parameters
function sayHello(hello) {
    console.log(hello);
}
sayHello("I love Umair");

// Function with Return Value
function sum(x, y) {
    let i = x + y;
    return i;
}
let val = sum(10, 20);
console.log(val);
