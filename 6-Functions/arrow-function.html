Arrow Functions in JavaScript

In this example, we demonstrate the use of arrow functions in JavaScript for performing basic arithmetic operations and counting vowels in a string.

Code Example

// Arrow Functions for Basic Arithmetic Operations

// Addition
const arrowAdd = (a, b) => {
    console.log("A + B =", a + b);
    return a + b;
};

// Multiplication
const arrowMul = (a, b) => {
    console.log("A * B =", a * b);
    return a * b;
};

// Subtraction
const arrowSub = (a, b) => {
    console.log("A - B =", a - b);
    return a - b;
};

// Division
const arrowDiv = (a, b) => {
    console.log("A / B =", a / b);
    return a / b;
};

// Counting Vowels in a String
function vouwals(str) {
    console.log(str);
    let count = 0;
    for (const key of str) {
        if ("aeiouAEIOU".includes(key)) {
            count++;
        }
    }
    console.log("Vowels Count:", count);
}
vouwals("ALI");

Output

A + B = 10
A * B = 20
A - B = 5
A / B = 2
Input String: ALI
Vowel Count: 2
